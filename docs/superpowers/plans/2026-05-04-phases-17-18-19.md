"""Load testing for concurrent users."""

import pytest
import asyncio
from fastapi.testclient import TestClient

@pytest.mark.asyncio
async def test_concurrent_retrieve_requests():
    """3 concurrent retrieve requests complete successfully."""
    from app.main import app
    import httpx

    # Use httpx.AsyncClient for async requests
    async with httpx.AsyncClient(app=app, base_url="http://test") as client:
        tasks = []

        async def retrieve(query: str):
            resp = await client.post("/api/v1/retrieve", json={
                "query": query,
                "kb_name": "default",
                "retrieval_pipeline": "tree",
                "top_k": 3,
            })
            return resp.status_code

        # 3 concurrent requests
        for i in range(3):
            tasks.append(retrieve(f"test query {i}"))

        results = await asyncio.gather(*tasks)

        # All should complete (200 or 500 if no data)
        assert all(r in (200, 500) for r in results)

@pytest.mark.asyncio
async def test_multiple_queries_timing():
    """5 queries complete within reasonable time."""
    from app.main import app
    import httpx
    import time

    async with httpx.AsyncClient(app=app, base_url="http://test") as client:
        start = time.time()

        tasks = []

                    "query": f"query {i}",
                    "kb_name": "default",
                    "top_k": 1,

@router.get("/health")
async def health_check():
    """System health check."""
    import shutil
    from app.main import lm_client, vram_monitor

    status = "healthy"
    deps = {}

    # LM Studio
    lm_ok = await lm_client.check_health()
    deps["lm_studio"] = "up" if lm_ok else "down"

    # GPU
    gpu_ok = vram_monitor.initialized
    deps["gpu_monitor"] = "up" if gpu_ok else "down"

    # Disk space
    try:
        total, used, free = shutil.disk_usage("/")
        free_gb = free / (1024 ** 3)
        deps["disk_space_gb_free"] = round(free_gb, 1)
        if free_gb < 1.0:
            status = "degraded"
    except Exception:
        deps["disk_space_gb_free"] = "unknown"

    # Overall status
    if not lm_ok:
        status = "degraded"

    return {"status": status, "dependencies": deps, "uptime_seconds": time.time() - _startup_time}

# Deployment Guide

## Docker Compose (Recommended)

### Prerequisites
- Docker Desktop with WSL2 (Windows) or Docker Engine (Linux)
- NVIDIA GPU with 8GB+ VRAM
- LM Studio installed with GGUF models

### Steps

1. Clone the repository:
   ```bash
   git clone https://github.com/shinshekai/Deep.git
   cd Deep
   ```

2. Create `.env` from example:
   ```bash
   cp .env.example .env
   ```

3. Edit `.env` with your settings:
   ```bash
   LLM_MODEL=Qwen3-4B-Q4_K_M
   LLM_HOST=http://host.docker.internal:1234
   BACKEND_PORT=8001
   FRONTEND_PORT=3782
   ```

4. Start services:
   ```bash
   docker-compose up --build -d
   ```

5. Check health:
   ```bash
   curl http://localhost:8001/api/v1/health
   ```

## Manual Deployment

### Backend

1. Create Python 3.10+ environment:
   ```bash
   cd backend
   python -m venv .venv
   source .venv/bin/activate  # or Scripts/activate on Windows
   pip install -e ".[dev]"
   ```

2. Configure environment:
   ```bash
   cp .env.example .env
   # Edit .env with your settings
   ```

3. Run with systemd (Linux):
   Create `/etc/systemd/system/udip-backend.service`:
   ```ini
   [Unit]
   Description=UDIP Backend
   After=network.target

   [Service]
   Type=simple
   User=youruser
   WorkingDirectory=/path/to/Deep/backend
   Environment="PATH=/path/to/Deep/backend/.venv/bin"
   ExecStart=/path/to/Deep/backend/.venv/bin/uvicorn app.main:app --host 0.0.0.0 --port 8001
   Restart=always

   [Install]
   WantedBy=multi-user.target
   ```

   Enable and start:
   ```bash
   sudo systemctl enable udip-backend
   sudo systemctl start udip-backend
   ```
